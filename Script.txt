CREATE TABLE tipos_de_aviones
(
cod_tipos_TA varchar(10),
descripcion_TA varchar(255) NOT NULL,
cant_pil_TA int NULL,
cant_jc_TA int NULL,
cant_com_TA int NULL,
cant_aux_TA int NULL,
CONSTRAINT PK_TA PRIMARY KEY (cod_tipos_TA)
)
GO

CREATE TABLE aviones
(
codigo_AV varchar (10) NOT NULL,
tipo_AV varchar (10) NOT NULL,
CONSTRAINT PK_AV PRIMARY KEY (codigo_AV)
)
GO

CREATE TABLE vuelos
(
numero_vuelo_VLO varchar(10) NOT NULL,
codigo_ruta_VLO varchar(10) NOT NULL,
horario_VLO smalldatetime NOT NULL,
CONSTRAINT PK_VLO PRIMARY KEY (numero_vuelo_VLO)
)
GO

CREATE TABLE rutas
(
codigo_RTA varchar(10),
ATOpartida_RTA varchar(10) NOT NULL,
ATOarrivo_RTA varchar(10) NOT NULL,
ETA_RTA time NOT NULL,
posta_RTA time NOT NULL,
CONSTRAINT PK_ATO PRIMARY KEY (codigo_RTA)
)
GO

CREATE TABLE rutas_por_TA
(
numero_VLO varchar(10) NOT NULL,
codigo_AV varchar(10) NOT NULL,
CONSTRAINT PK_VLO_POR_AV PRIMARY KEY (numero_VLO, codigo_AV)
)
GO

CREATE TABLE Tipos_De_Tripulantes
(
codigo_TT varchar(10) NOT NULL,
nombre_TT varchar (50) NOT NULL,
descripcion_TT varchar (50) NOT NULL,
Constraint PK_TT PRIMARY KEY (codigo_TT)
)
GO

CREATE TABLE tripulantes
(
legajo_TCP varchar (10) NOT NULL,

nombre_TCP varchar (20) NOT NULL,
apellido_TCP varchar (20) NOT NULL,
cuil_TCP varchar (10) NOT NULL,
telefono_TCP varchar (20) NOT NULL,
CONSTRAINT PK_TCP PRIMARY KEY (legajo_TCP)
)
GO
